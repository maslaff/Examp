y = k1 * x + b1, y = k2 * x + b2

y = 5 * x + 2, y = 9 * x + 4

k1 * x + b1 = k2 * x + b2
 5 * x + 2  = 9  * x + 4

(k1 - k2) * x = b2 - b1
 5  - 9   * x = 4  - 2

x = (b2 - b1) / (k1 - k2)
x = (4  - 2)  / (5  - 9)
x =     2     /    -4       // -0,5

y = k1 *  x   + b1
y = 5  *  x   + 2
y = 5  * -0,5 + 2              // -0,5
